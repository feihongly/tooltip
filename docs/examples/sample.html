How to use

Include


<link rel="stylesheet" type="text/css" href="/dist/css/gf.tooltip.css">
<script src="/dist/plugin/jquery.min.js"></script>
<script src="/dist/js/jquery.gf.tooltip.min.js"></script>

HTML


<span class="tooltip" data-tips="Here is tooltips">Move cursor to here</span>

JS


$(function(){
    $('body').gfTooltip({
        target: '.tooltip',
        contents: function(elem){
            return $(elem).attr('data-tips');
        }
    });
});

About Settings

Settings


//可以将tooltip绑定到任何已经存在的元素上
//建议绑定到已有的元素，然后target设定为需要显示tooltip的元素
//target可以是ready时候并不存在的元素，这样就可以支持实时生成出来的元素了
$('body').gfTooltip({
    target: '.tooltip',
    //offset表示tooltip的偏移量
    offset: {
        x: 20,
        y: 10
    },
    //contents允许string，也支持function返回
    //其中elem为当前target的jquery对象
    //可以在回调函数中，对elem进行操作
    //例如通过获取elem对象的属性，来return需要显示的tips
    contents: function(elem){
        return $(elem).attr('data-tips');
    }
});
