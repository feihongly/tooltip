How to use

Include


<link rel="stylesheet" type="text/css" href="/dist/css/gf.tooltip.css">
<script src="/dist/plugin/jquery.min.js"></script>
<script src="/dist/js/jquery.gf.tooltip.min.js"></script>

HTML


<span class="tooltip" data-tips="Here is tooltips">Move cursor to here</span>

JS


$(function(){
    $('body').gfTooltip({
        target: '.tooltip',
        autoPosition: true,
        contents: function(elem){
            return $(elem).attr('data-tips');
        }
    });
});
